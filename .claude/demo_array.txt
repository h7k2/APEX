let favorites = JSON.parse(localStorage.getItem('apex_favorites')||'[]');

// ── API ─────────────────────────────────────────────────────────
async function apiFetch(path, opts={}) {
  const headers = {'Content-Type':'application/json',...opts.headers};
  if (authToken) headers['Authorization'] = `Bearer ${authToken}`;
  const res = await fetch(API+path, {...opts, headers});
  if (!res.ok) { const e=await res.json().catch(()=>({})); throw new Error(e.detail||`Erreur ${res.status}`); }
  return res.json();
}

// ── INIT ────────────────────────────────────────────────────────
async function init() {
  updateAuthUI();
  updateCart();
  updateFavBadge();
  try {
    const data = await apiFetch('/products');
    apiOnline = true;
    if (data.length > 0) {
      allProducts = data;
      setBanner('online', `✓ API connectée — ${data.length} produits depuis la base de données`);
      document.getElementById('statProducts').textContent = data.length+'+';
    } else {
      allProducts = DEMO;
      setBanner('online', `✓ API connectée — base vide, données de démo affichées`);
      document.getElementById('statProducts').textContent = DEMO.length+'+';
    }
  } catch(e) {
    allProducts = DEMO; apiOnline = false;
    setBanner('offline', '⚠ API hors ligne — données de démo affichées. Lancez : uvicorn main:app --reload');
    document.getElementById('statProducts').textContent = DEMO.length+'+';
  }
  renderSections();
}

function setBanner(type, msg) {
  const b = document.getElementById('apiBanner');
  b.style.display='flex'; b.className=`api-banner ${type}`;
  document.getElementById('apiBannerText').textContent = msg;
}

// ── FAVORIS ────────────────────────────────────────────────────
function toggleFavorite(id, btn) {
  if (!currentUser) {
    showToast('Connectez-vous pour ajouter aux favoris', true);
    openModal('authModal');
    return;
  }
  const idx = favorites.indexOf(id);
  if (idx > -1) {
    favorites.splice(idx, 1);
    if(btn){ btn.classList.remove('favorited'); btn.textContent='♡'; }
    showToast('Retiré des favoris');
  } else {
    favorites.push(id);
    if(btn){ btn.classList.add('favorited'); btn.textContent='♥'; }
    const p = allProducts.find(p=>p.id===id);
    showToast(`${p?.emoji||'♥'} ${p?.name||'Produit'} ajouté aux favoris !`);
  }
  localStorage.setItem('apex_favorites', JSON.stringify(favorites));
  updateFavBadge();
}

function updateFavBadge() {
  const btn = document.getElementById('favBtn');
  const count = favorites.length;
  let badge = btn.querySelector('.fav-badge');
  if (count > 0) {
    if (!badge) {
      badge = document.createElement('span');
      badge.className = 'cart-badge fav-badge';
      btn.appendChild(badge);
    }
    badge.textContent = count;
    badge.style.display = 'flex';
  } else if (badge) {
    badge.style.display = 'none';
  }
}

// ── PANIER ──────────────────────────────────────────────────────
function addToCart(id) {
  const p = allProducts.find(p=>p.id===id); if(!p) return;
  const ex = cart.find(i=>i.id===id);
  if (ex) { if(ex.qty>=p.stock){showToast('⚠ Stock max atteint',true);return;} ex.qty++; }
  else cart.push({...p,qty:1});
  updateCart(); showToast(`${p.emoji} ${p.name} ajouté !`);
}

function removeFromCart(id) { cart=cart.filter(i=>i.id!==id); updateCart(); }

function changeQty(id,d) {
  const item=cart.find(i=>i.id===id); if(!item) return;
  item.qty+=d;
  if(item.qty<=0) removeFromCart(id); else updateCart();
}

function updateCart() {
  const total=cart.reduce((s,i)=>s+i.price*i.qty,0);
  const count=cart.reduce((s,i)=>s+i.qty,0);
  const badge=document.getElementById('cartCount');
  badge.textContent=count;
  badge.style.display=count>0?'flex':'none';
  document.getElementById('cartTotal').textContent=fmt(total);
  document.getElementById('checkoutBtn').disabled=cart.length===0;
  const c=document.getElementById('cartItems');
  if(!cart.length){ c.innerHTML=`<div class="cart-empty"><div class="cart-empty-icon">🛒</div><p>Votre panier est vide.</p></div>`; return; }
  c.innerHTML=cart.map(item=>`
    <div class="cart-item">
      <div class="cart-item-emoji">${item.emoji||'📦'}</div>
      <div class="cart-item-info">
        <div class="cart-item-name">${item.name}</div>
        <div class="cart-item-price">${fmt(item.price*item.qty)}</div>
        <div class="cart-item-qty">
          <button class="qty-btn" onclick="changeQty(${item.id},-1)">−</button>
          <span class="qty-val">${item.qty}</span>
          <button class="qty-btn" onclick="changeQty(${item.id},+1)">+</button>
        </div>
      </div>
      <button class="remove-btn" onclick="removeFromCart(${item.id})">✕</button>
    </div>`).join('');
}

// ── AUTH ────────────────────────────────────────────────────────
function handleAuthClick() {
  if(currentUser){ if(confirm(`Déconnexion de ${currentUser.full_name} ?`)) logout(); }
  else openModal('authModal');
}

function logout() {
  currentUser=null; authToken=null;
  localStorage.removeItem('apex_user'); localStorage.removeItem('apex_token');
  updateAuthUI(); showToast('Déconnecté');
}

function updateAuthUI() {
  const btn=document.getElementById('authBtn');
  const dot=document.getElementById('authDot');
  const logged=!!currentUser;
  btn.classList.toggle('active-user',logged);
  dot.classList.toggle('visible',logged);
}

async function login() {
  const email=document.getElementById('loginEmail').value.trim();
  const pw=document.getElementById('loginPassword').value;
  const err=document.getElementById('loginError');
  const btn=document.getElementById('loginBtn');
  err.classList.remove('show');
  if(!email||!pw){showErr(err,'Veuillez remplir tous les champs');return;}
  btn.disabled=true; btn.textContent='Connexion…';
  try {
    const form=new URLSearchParams({username:email,password:pw});
    const res=await fetch(`${API}/auth/login`,{method:'POST',headers:{'Content-Type':'application/x-www-form-urlencoded'},body:form});
    if(!res.ok){const e=await res.json();throw new Error(e.detail);}
    const data=await res.json();
    authToken=data.access_token; currentUser=data.user;
    localStorage.setItem('apex_token',authToken);
    localStorage.setItem('apex_user',JSON.stringify(currentUser));
    updateAuthUI(); closeModal('authModal');
    showToast(`Bienvenue ${currentUser.full_name} !`);
  } catch(e){ showErr(err,e.message||'Erreur de connexion'); }
  finally{ btn.disabled=false; btn.textContent='Se connecter'; }
}

async function register() {
  const name=document.getElementById('regName').value.trim();
  const email=document.getElementById('regEmail').value.trim();
  const pw=document.getElementById('regPassword').value;
  const err=document.getElementById('registerError');
  const btn=document.getElementById('registerBtn');
  err.classList.remove('show');
  if(!name||!email||!pw){showErr(err,'Veuillez remplir tous les champs');return;}
  if(pw.length<8){showErr(err,'Mot de passe : 8 caractères minimum');return;}
  btn.disabled=true; btn.textContent='Création…';
  try {
    await apiFetch('/auth/register',{method:'POST',body:JSON.stringify({full_name:name,email,password:pw})});
    showToast('Compte créé ! Email de bienvenue envoyé.');
    document.getElementById('loginEmail').value=email;
    document.getElementById('loginPassword').value=pw;
    switchTab('login'); await login();
  } catch(e){ showErr(err,e.message||'Erreur de création'); }
  finally{ btn.disabled=false; btn.textContent='Créer mon compte'; }
}

// ── COMMANDE ────────────────────────────────────────────────────
function checkout() {
  if(!currentUser){ showToast('Connectez-vous pour commander',true); openModal('authModal'); return; }
  toggleCart();
  const total=cart.reduce((s,i)=>s+i.price*i.qty,0);
  document.getElementById('checkoutSummary').innerHTML=cart.map(i=>
    `<div class="recap-item"><span>${i.emoji} ${i.name} ×${i.qty}</span><span>${fmt(i.price*i.qty)}</span></div>`
  ).join('');
  document.getElementById('checkoutTotalDisplay').textContent=`Total : ${fmt(total)}`;
  openModal('checkoutModal');

async function confirmOrder() {
  const address=document.getElementById('orderAddress').value.trim();
  const err=document.getElementById('checkoutError');
  const btn=document.getElementById('confirmOrderBtn');
  err.classList.remove('show');
  if(!address){showErr(err,'Veuillez entrer une adresse');return;}
  btn.disabled=true; btn.textContent='Confirmation…';
  try {
    const order=await apiFetch('/orders',{method:'POST',body:JSON.stringify({address,items:cart.map(i=>({product_id:i.id,quantity:i.qty}))})});
    cart=[]; updateCart(); closeModal('checkoutModal');
    showToast(`Commande #${order.id} confirmée !`);
  } catch(e) {
    if(!apiOnline){
      cart=[]; updateCart(); closeModal('checkoutModal');
      showToast('Commande simulée avec succès !');
    } else { showErr(err,e.message||'Erreur commande'); }
  }
  finally{ btn.disabled=false; btn.textContent='Confirmer la commande'; }
}

function openPdp(id) {

  pdpCameFromCategory = document.getElementById('categoryView').style.display==='block';

  // Breadcrumb

  // Infos

  document.getElementById('pdpNameEl').textContent = p.name;
  document.getElementById('pdpPriceEl').textContent = fmt(p.price);
  document.getElementById('pdpPriceOrig').textContent = p.original_price ? fmt(p.original_price) : '';

    document.querySelector('#pdpSizesBlock .pdp-section-label').textContent = p.variants.label;

    sizesBlock.style.display = 'none';

  setupGallery(p, 0);

  renderReviews(p.id);

  window.scrollTo(0,0);
  document.getElementById('catalogView').style.display = 'none';
  document.getElementById('categoryView').style.display = 'none';
  document.getElementById('pdpView').style.display = 'block';
}

function setupGallery(p, colorIndex) {
  const gallery = document.getElementById('pdpGallery');

  let images = [];
  if (p.colors && p.colors.length > 0) {
    images = p.colors[colorIndex].images;
  } else if (p.images) {
    images = p.images;
  } else if (p.image) {
    images = [p.image];
  }

  if (images.length > 1) {
    gallery.innerHTML = `
      <div class="pdp-gallery-with-thumbs">
        <div class="pdp-thumbs-vertical" id="pdpThumbsV">
          ${images.map((src,i)=>`
            <div class="pdp-thumb${i===0?' active':''}" onmouseenter="pdpHoverThumb('${src}',this)" onclick="openLightbox('${src}')">
              <img src="${src}" alt=""></div>`).join('')}
        </div>
        <div class="pdp-gallery-main" id="pdpGalleryMain">
          <img src="${images[0]}" alt="${p.name}" onclick="openLightbox(this.src)">
        </div>
      </div>`;
  } else if (images.length === 1) {
    gallery.innerHTML = `
      <div class="pdp-main-img">
        <img src="${images[0]}" alt="${p.name}" onclick="openLightbox(this.src)">
      </div>`;
  } else {
    gallery.innerHTML = `
      <div class="pdp-main-img"><div class="pdp-emoji">${p.emoji||'📦'}</div></div>`;
  }
}

function selectColor(index) {
  if (!currentPdpProduct || !currentPdpProduct.colors) return;
  document.querySelectorAll('.color-swatch').forEach((s,i)=>s.classList.toggle('active',i===index));
  document.getElementById('pdpColorName').textContent = currentPdpProduct.colors[index].name;
  setupGallery(currentPdpProduct, index);
}

function pdpHoverThumb(src, thumb) {
  const main = document.getElementById('pdpGalleryMain');
  if (main) {
    main.innerHTML = `<img src="${src}" alt="" onclick="openLightbox(this.src)">`;
  }
  document.querySelectorAll('.pdp-thumbs-vertical .pdp-thumb').forEach(t=>t.classList.remove('active'));
  thumb.classList.add('active');
}

function renderReviews(productId) {
  const container = document.getElementById('pdpReviews');
  const reviews = REVIEWS[productId];
  if (!reviews || reviews.length === 0) {
    container.innerHTML = '';
    return;
  }
  const avg = (reviews.reduce((s,r) => s + r.rating, 0) / reviews.length).toFixed(1);
  const fullStars = Math.floor(avg);
  const hasHalf = avg - fullStars >= 0.3;
  const starsHtml = '★'.repeat(fullStars) + (hasHalf ? '½' : '') + '☆'.repeat(5 - fullStars - (hasHalf?1:0));

  container.innerHTML = `
    <div class="pdp-reviews-header">
      <div class="pdp-reviews-title">Avis clients (${reviews.length})</div>
      <div class="pdp-reviews-avg">
        <span class="pdp-reviews-stars">${starsHtml}</span>
        <span>${avg}/5</span>
      </div>
    </div>
    ${reviews.map(r => `
      <div class="review-card">
        <div class="review-top">
          <div>
            <span class="review-author">${r.name}</span>
            <span class="review-verified">✓ Achat vérifié</span>
          </div>
          <span class="review-date">${r.date}</span>
        </div>
        <div class="review-stars">${'★'.repeat(r.rating)}${'☆'.repeat(5-r.rating)}</div>
        <div class="review-text">${r.text}</div>
        ${r.images ? `<div class="review-images">${r.images.map(img => `<img src="${img}" alt="Photo client" onclick="event.stopPropagation();openLightbox('${img}')">`).join('')}</div>` : ''}
      </div>
    `).join('')}
  `;
}

}
